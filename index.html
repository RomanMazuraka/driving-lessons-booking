<script>


let studentView,adminView,exitBtn;

let fullnameInput,phoneInput,emailInput,dateInput,timeSelect;

let requestsDiv,scheduleGrid,calendarEl;


let db=null;


const firebaseConfig={

apiKey:"PASTE_KEY",

authDomain:"PASTE",

databaseURL:"PASTE",

projectId:"PASTE",

storageBucket:"PASTE",

messagingSenderId:"PASTE",

appId:"PASTE"

};


try{

if(firebaseConfig.apiKey!=="PASTE_KEY"){

firebase.initializeApp(firebaseConfig);

db=firebase.database();

}

}catch(e){console.warn("Firebase disabled",e)}


/* ADMIN LOGIN */


let ADMIN_PASSWORD="1234";


function adminLogin(){


const passInput=document.getElementById("adminPass");

const loginBtn=document.getElementById("loginBtn");


passInput.style.display="inline-block";

loginBtn.style.display="inline-block";


loginBtn.onclick=()=>{


const pass=passInput.value;


if(pass===ADMIN_PASSWORD){


studentView.style.display="none";

adminView.style.display="block";

exitBtn.style.display="inline-block";


passInput.style.display="none";

loginBtn.style.display="none";


buildSchedule();

render();


}else{

alert("Wrong password");

}


};


}


function exitAdmin(){

studentView.style.display="grid";

adminView.style.display="none";

exitBtn.style.display="none";

}


/* LANGUAGE */


let lang="uk";


const translations={

uk:{booking:"Запис на урок",name:"Ім'я та прізвище",phone:"Телефон",date:"Дата",time:"Час",book:"Записатись",requests:"Заявки",availabilityNote:"Показуються тільки доступні години.",calendar:"Календар"},

cs:{booking:"Rezervace jízdy",name:"Jméno a příjmení",phone:"Telefon",date:"Datum",time:"Čas",book:"Rezervovat",requests:"Žádosti",availabilityNote:"Zobrazují se pouze dostupné hodiny.",calendar:"Kalendář"}

};


function setLang(l){lang=l;applyLang();}


function applyLang(){

const t=translations[lang];


document.getElementById("bookingTitle").innerText=t.booking;

document.getElementById("labelName").innerText=t.name;

document.getElementById("labelPhone").innerText=t.phone;

document.getElementById("labelDate").innerText=t.date;

document.getElementById("labelTime").innerText=t.time;

document.getElementById("bookBtn").innerText=t.book;

document.getElementById("requestsTitle").innerText=t.requests;


document.getElementById("availabilityNote").innerText=t.availabilityNote;

document.getElementById("calendarTitle").innerText=t.calendar;


render();

buildCalendar();

}


/* DATA */


let availability={};

let requests={};


// generate time slots every 30 minutes from 07:00 to 20:00

const hours=[];

for(let h=7;h<=20;h++){

let hh=h.toString().padStart(2,'0');

hours.push(`${hh}:00`);

if(h!==20) hours.push(`${hh}:30`);

}

const days=["Mon","Tue","Wed","Thu","Fri","Sat","Sun"];


function loadData(){


if(!db){

availability=JSON.parse(localStorage.getItem("availability")||"{}");

requests=JSON.parse(localStorage.getItem("requests")||"{}");

buildSchedule();

buildCalendar();

render();

return;

}


 db.ref("availability").on("value",snap=>{

 availability=snap.val()||{};

 buildSchedule();

 buildCalendar();

 updateTimeSlots();

 });


 db.ref("requests").on("value",snap=>{

 requests=snap.val()||{};

 render();

 buildCalendar();

 updateTimeSlots();

 });


}


function isBooked(day,hour){

for(const id in requests){

const r=requests[id];

const d=(new Date(r.date).getDay()+6)%7;

if(d===day && r.time===hour && r.status!=="cancel") return true;

}

return false;

}


function buildSchedule(){

if(!scheduleGrid) return;


scheduleGrid.innerHTML="";


scheduleGrid.appendChild(document.createElement("div"));


days.forEach(d=>{

const el=document.createElement("div");

el.className="day";

el.innerText=d;

scheduleGrid.appendChild(el);

});


hours.forEach(h=>{


const hour=document.createElement("div");

hour.innerText=h;

scheduleGrid.appendChild(hour);



days.forEach((d,i)=>{


const slot=document.createElement("div");

slot.className="slot";


const key=i+"_"+h;


if(availability[key]) slot.classList.add("active");

if(isBooked(i,h)) slot.classList.add("booked");


slot.onclick=()=>{


if(isBooked(i,h)){

alert("Slot already booked");

return;

}


if(availability[key]) delete availability[key];

else availability[key]=true;


 if(db){

 db.ref("availability").set(availability);

}else{

 localStorage.setItem("availability",JSON.stringify(availability));

 buildSchedule();

 buildCalendar();

}


};


scheduleGrid.appendChild(slot);


});


});

}


function updateTimeSlots(){


if(!dateInput) return;


const date=dateInput.value;


if(!date) return;


const day=(new Date(date).getDay()+6)%7;


timeSelect.innerHTML="";


hours.forEach(h=>{


if(availability[day+"_"+h] && !isBooked(day,h)){

const opt=document.createElement("option");

opt.value=h;

opt.textContent=h;

timeSelect.appendChild(opt);

}


});


}


function bookLesson(){


const fullname=fullnameInput.value.trim();

const phone=phoneInput.value.trim();

const email=emailInput.value.trim();

if(!/^\d{9}$/.test(phone)){

alert("Telefon musí mít 9 číslic (např. 777123456)");

return;

}


if(!/^\S+@\S+\.\S+$/.test(email)){

alert("Zadejte platný email");

return;

}

const date=dateInput.value;

const time=timeSelect.value;


if(!fullname||!phone||!email||!date||!time){

alert("Fill all fields");

return;

}


const id=Date.now();


if(db){

 db.ref("requests/"+id).set({fullname,phone,email,date,time,status:"pending"});

}else{

 requests[id]={fullname,phone,email,date,time,status:"pending"};

 localStorage.setItem("requests",JSON.stringify(requests));

 render();

 buildCalendar();

}


alert("Booking request sent");


}


function confirmLesson(id){

if(db){

 db.ref("requests/"+id+"/status").set("confirmed");

}else{

 requests[id].status="confirmed";

 localStorage.setItem("requests",JSON.stringify(requests));

 render();

 buildCalendar();

}

}


function cancelLesson(id){

if(db){

 db.ref("requests/"+id+"/status").set("cancel");

}else{

 requests[id].status="cancel";

 localStorage.setItem("requests",JSON.stringify(requests));

 render();

 buildCalendar();

}

}


function render(){


if(!requestsDiv) return;


requestsDiv.innerHTML="";


for(const id in requests){


const r=requests[id];


const div=document.createElement("div");

div.className="lesson";


let html=`<b>${r.fullname}</b><br>${r.phone}<br>${r.email}<br>${r.date} ${r.time}<br><div class="status">${r.status}</div>`;


if(r.status==="pending"){

html+=`<button class="confirm" onclick="confirmLesson('${id}')">Confirm</button>

<button onclick="cancelLesson('${id}')" style="background:#c0392b">Cancel</button>`;

}


if(r.status==="confirmed"){

html+=`<button onclick="cancelLesson('${id}')" style="background:#c0392b">Cancel</button>`;

}



div.innerHTML=html;

requestsDiv.appendChild(div);


}


}


function buildCalendar(){


if(!calendarEl) return;


calendarEl.innerHTML="";


for(let d=0;d<7;d++){


const dayBox=document.createElement("div");

dayBox.className="calendar-day";


const title=document.createElement("h4");

title.innerText=days[d];


dayBox.appendChild(title);


hours.forEach(h=>{


if(availability[d+"_"+h]){


const slot=document.createElement("div");

slot.className="calendar-slot";


if(isBooked(d,h)){

slot.classList.add("slot-booked");

slot.innerText=h+" booked";

}else{

slot.classList.add("slot-free");

slot.innerText=h;

}



dayBox.appendChild(slot);


}


});


calendarEl.appendChild(dayBox);


}


}

studentView=document.getElementById("studentView");

adminView=document.getElementById("adminView");

exitBtn=document.getElementById("exitBtn");


fullnameInput=document.getElementById("fullname");

phoneInput.value=phoneInput.value.replace(/[^0-9]/g,"");

});


dateInput=document.getElementById("date");

timeSelect=document.getElementById("time");


requestsDiv=document.getElementById("requests");

scheduleGrid=document.getElementById("scheduleGrid");

calendarEl=document.getElementById("calendar");


loadData();


const adminBtn=document.getElementById("adminBtn");

if(adminBtn){adminBtn.addEventListener("click",adminLogin)}


applyLang();


});
